# 3. 本地数据解析器 (处理 zip 和手动数据)
# ==========================================
class LocalDataProcessor:
    def __init__(self):
        self.search_root = CONFIG['extract_dir']

    def find_best_file(self, keywords):
        candidates = []
        for root, dirs, files in os.walk(self.search_root):
            if any(x in root.lower() for x in ['__pycache__', '.git']): continue
            # 路径匹配
            path_match = any(k in os.path.basename(root).lower() for k in keywords)

            for f in files:
                f_lower = f.lower()
                if f_lower.endswith(('.jsonl', '.json', '.csv', '.parquet')):
                    # 排除索引文件
                    if 'ids' in f_lower or 'idx' in f_lower: continue

                    priority = 0
                    if path_match: priority += 5
                    if any(k in f_lower for k in keywords): priority += 3
                    if 'test' in f_lower: priority += 2
                    elif 'val' in f_lower: priority += 1

                    if priority > 0:
                        candidates.append((priority, os.path.join(root, f)))

        candidates.sort(key=lambda x: x[0], reverse=True)
        return candidates[0][1] if candidates else None

    def normalize_entry(self, entry, bench_name=""):
        if not isinstance(entry, dict): return None
        # 转小写Key方便查找
        keys = {k.lower(): k for k in entry.keys()}

        # === 特殊格式处理 ===
        # 1. CPED / 情感
        if 'cped' in bench_name.lower():
            u_key = next((k for k in ['utterance', 'sentence', 'text'] if k in keys), None)
            e_key = next((k for k in ['emotion', 'label', 'sentiment'] if k in keys), None)
            if u_key and e_key:
                return {"prompt": f"Analyze the emotion.\nSentence: \"{entry[keys[u_key]]}\"\nAnswer:", "gold": str(entry[keys[e_key]])}

        # 2. ToMATO / 心智理论
        if 'tomato' in bench_name.lower() or ('conversation' in keys and 'a0' in keys):
            ctx = entry.get(keys.get('conversation', ''), '') or entry.get(keys.get('story', ''), '')
            q = entry.get(keys.get('q', ''), '')
            if not q: return None
            # 构造选项
            opts = []
            for i in range(5):
                k = f'a{i}'
                if k in keys: opts.append(f"{chr(65+i)}. {entry[keys[k]]}")
            opts_str = "\nOptions:\n" + "\n".join(opts)
            gold_idx = entry.get(keys.get('a_idx', ''), '')
            gold = chr(65 + int(gold_idx)) if str(gold_idx).isdigit() else str(entry.get(keys.get('a_str', ''), ''))
            return {"prompt": f"Context:\n{ctx}\n\nQuestion: {q}{opts_str}\nAnswer:", "gold": gold}

        # === 通用格式匹配 ===
        # 找 Context
        context = ""
        ctx_key = next((k for k in ['story', 'context', 'passage', 'dialogue'] if k in keys and isinstance(entry[keys[k]], str) and len(entry[keys[k]]) > 50), None)
        if ctx_key: context = f"Context:\n{entry[keys[ctx_key]]}\n\n"

        # 找 Question
        q_cands = ['question', 'input', 'prompt', 'problem', 'text', 'input_text']
        if ctx_key in q_cands: q_cands.remove(ctx_key)
        q_key = next((k for k in q_cands if k in keys), None)
        if not q_key: return None

        # 找 Options (MCQ)
        opts_str = ""
        opt_key = next((k for k in ['options', 'choices'] if k in keys), None)
        if opt_key:
            opts = entry[keys[opt_key]]
            if isinstance(opts, list): opts_str = "\nOptions:\n" + "\n".join([f"{i}. {o}" for i, o in enumerate(opts)])
            elif isinstance(opts, dict): opts_str = "\nOptions:\n" + "\n".join([f"{k}. {v}" for k, v in opts.items()])

        # 找 Answer
        a_key = next((k for k in ['answer', 'label', 'gold', 'output', 'target', 'solution', 'canonical_solution'] if k in keys), None)
        if not a_key: return None

        gold = str(entry[keys[a_key]])
        if gold.startswith("['") and gold.endswith("']"): gold = gold[2:-2]

        return {"prompt": f"{context}Question: {entry[keys[q_key]]}{opts_str}\nAnswer:", "gold": gold}

    def load_data(self, bench_name, keywords):
        raw_file = self.find_best_file(keywords)
        if not raw_file: return []

        print(f"     📂 [Local] Found file: {os.path.basename(raw_file)}")
        data = []
        entries = []
        try:
            if raw_file.endswith('.csv'):
                try: entries = pd.read_csv(raw_file).to_dict('records')
                except: pass
            elif raw_file.endswith('.jsonl') or raw_file.endswith('.json'):
                with open(raw_file, 'r', encoding='utf-8') as f:
                    if raw_file.endswith('.jsonl'):
                        for line in f:
                            try: entries.append(json.loads(line))
                            except: pass
                    else:
                        content = json.load(f)
                        entries = content if isinstance(content, list) else (content['data'] if 'data' in content else [])

            # 特殊处理 parquet
            elif raw_file.endswith('.parquet'):
                entries = pd.read_parquet(raw_file).to_dict('records')

            for entry in entries:
                processed = self.normalize_entry(entry, bench_name)
                if processed: data.append(processed)
        except Exception as e:
            print(f"     ❌ Local parse error: {e}")

        return data

# 4. 混合数据加载器 (Hybrid Loader)

class HybridDataLoader:
    def __init__(self):
        self.hf_root = CONFIG['raw_data_dir']
        self.local_root = CONFIG['local_data_dir']
        self.local_processor = LocalDataProcessor()

        if not os.path.exists(self.hf_root): os.makedirs(self.hf_root)
        if not os.path.exists(self.local_root): os.makedirs(self.local_root)

    def unzip_local_files(self):
        if not os.path.exists(CONFIG['extract_dir']): os.makedirs(CONFIG['extract_dir'])
        zips = glob.glob(os.path.join(self.local_root, "*.zip"))
        if zips:
            print(f"\n📦 正在解压 {len(zips)} 个本地文件...")
            for z in zips:
                try:
                    with zipfile.ZipFile(z, 'r') as zf: zf.extractall(CONFIG['extract_dir'])
                except: pass

    def load_all(self):
        from huggingface_hub import snapshot_download, login
        try: login(token=CONFIG['hf_token'])
        except: pass

        self.unzip_local_files()
        print("\n🔄 开始加载数据集 (优先 HF -> 失败则回退 Local)...")

        tasks = []

        for name, meta in DATASETS_CFG.items():
            repo_id = meta['hf']
            keywords = meta['keys']
            hf_dir = os.path.join(self.hf_root, name)

            dataset_content = []
            source = "none"

            # --- 步骤 1: 尝试从 HF 加载/下载 ---
            try:
                # 检查本地缓存或下载
                if not os.path.exists(hf_dir) or not os.listdir(hf_dir):
                    print(f"  ☁️  Downloading HF: {name} ({repo_id})...")
                    snapshot_download(repo_id=repo_id, repo_type="dataset", local_dir=hf_dir,
                                      token=CONFIG['hf_token'], max_workers=4,
                                      ignore_patterns=["*.md", "original/*", ".git*"])

                # 尝试用标准逻辑读取 HF 文件
                # (复用之前的逻辑: 找 parquet/jsonl)
                files = glob.glob(os.path.join(hf_dir, "**/*"), recursive=True)
                valid_files = [f for f in files if f.endswith(('.parquet', '.jsonl', '.json', '.csv'))]

                # 优先级排序
                def file_prio(f):
                    return 0 if 'test' in f.lower() else (1 if 'val' in f.lower() else 2)
                valid_files.sort(key=file_prio)

                if valid_files:
                    target_f = valid_files[0]
                    if target_f.endswith('.parquet'): dataset_content = pd.read_parquet(target_f).to_dict('records')
                    elif target_f.endswith('.jsonl'):
                        with open(target_f, 'r', encoding='utf-8') as f: dataset_content = [json.loads(l) for l in f]
                    elif target_f.endswith('.json'):
                         with open(target_f, 'r', encoding='utf-8') as f: c=json.load(f); dataset_content=c if isinstance(c,list) else []

                    if dataset_content: source = "HuggingFace"

            except Exception as e:
                # 404/401 等错误会到这里
                if "404" in str(e) or "401" in str(e):
                    pass # 也就是没下载到，后面会走 Local

            # --- 步骤 2: 如果 HF 失败，尝试 Local ---
            if not dataset_content:
                print(f"     ⚠️ HF 加载失败/无数据，尝试读取本地 ./datasets ({keywords})...")
                dataset_content = self.local_processor.load_data(name, keywords)
                if dataset_content: source = "Local Disk"

            # --- 步骤 3: 标准化数据 ---
            if dataset_content:
                # 如果是 HF 来的数据，可能还没归一化，需要简单处理
                # 如果是 Local 来的，LocalDataProcessor 已经归一化过了
                final_data = []

                if source == "Local Disk":
                    final_data = dataset_content # 已经格式化好了
                else:
                    # 对 HF 数据做简单归一化 (复用 LocalProcessor 的逻辑来保证一致性)
                    for item in dataset_content:
                        processed = self.local_processor.normalize_entry(item, name)
                        if processed: final_data.append(processed)

                if final_data:
                    # 采样
                    if name == "MMLU_Pro":
                        cnt = max(1, int(len(final_data)*CONFIG['mmlu_sample_ratio']))
                        final_data = random.sample(final_data, cnt)

                    if CONFIG['test_limit']:
                        final_data = final_data[:CONFIG['test_limit']]

                    tasks.append((name, final_data))
                    print(f"  ✅ Ready: {name:<15} | Source: {source:<10} | Count: {len(final_data)}")
                else:
                    print(f"  ❌ Empty: {name} (Data found but parse failed)")
            else:
                print(f"  ❌ Missing: {name} (Download failed & No local file found)")

        return tasks

class Scorer:
    @staticmethod
    def normalize(text): return re.sub(r'\s+', ' ', str(text).lower().strip())
    @staticmethod
    def extract_answer(pred):
        # 尝试提取 "Answer: X" 或 "Boxed{X}"
        match = re.search(r'\\boxed\{([^}]+)\}', pred)
        if match: return match.group(1)
        match = re.search(r'(?:^|\s)([A-D])(?:\)|\.|,|$)', pred.upper()) # MCQ
        if match: return match.group(1)
        return pred

    @staticmethod
    def score(pred, gold):
        p_norm, g_norm = Scorer.normalize(pred), Scorer.normalize(gold)
        p_ext = Scorer.normalize(Scorer.extract_answer(pred))

        # 1. 精确匹配
        if p_ext == g_norm: return 1.0
        # 2. 包含匹配
        if g_norm in p_norm: return 1.0
        # 3. 拒绝检测 (XSTest)
        refusals = ["sorry", "cannot", "unable", "illegal"]
        is_refused = any(w in p_norm for w in refusals)
        if "refus" in g_norm: return 1.0 if is_refused else 0.0

        return 0.0

class APIEvaluator:
    def __init__(self, conf):
        self.client = OpenAI(api_key=conf['api_key'], base_url=conf['base_url'])
        self.model = conf['model_name']
        self.concurrency = conf.get('concurrency', 5)
    def generate(self, prompts):
        def _call(p):
            try: return self.client.chat.completions.create(model=self.model, messages=[{"role":"user","content":p}], max_tokens=CONFIG['max_new_tokens']).choices[0].message.content
            except: return ""
        with ThreadPoolExecutor(max_workers=self.concurrency) as ex: return list(ex.map(_call, prompts))
    def unload(self): pass

class LocalEvaluator:
    def __init__(self, conf):
        self.device = CONFIG['device']
        if HAS_UNSLOTH:
            try:
                self.model, self.tokenizer = FastLanguageModel.from_pretrained(model_name=conf['path'], max_seq_length=4096, load_in_4bit=True)
                FastLanguageModel.for_inference(self.model)
            except: self._load_hf(conf)
        else: self._load_hf(conf)
        if not self.tokenizer.pad_token: self.tokenizer.pad_token = self.tokenizer.eos_token
    def _load_hf(self, conf):
        self.tokenizer = AutoTokenizer.from_pretrained(conf['path'], padding_side='left', trust_remote_code=True)
        self.model = AutoModelForCausalLM.from_pretrained(conf['path'], device_map="auto", torch_dtype=torch.float16, trust_remote_code=True)
    def generate(self, prompts):
        inputs = self.tokenizer(prompts, return_tensors="pt", padding=True, truncation=True, max_length=2048).to(self.device)
        with torch.no_grad():
            outputs = self.model.generate(**inputs, max_new_tokens=CONFIG['max_new_tokens'], pad_token_id=self.tokenizer.pad_token_id, do_sample=False)
        return [self.tokenizer.decode(o[inputs.input_ids.shape[1]:], skip_special_tokens=True) for o in outputs]
    def unload(self): del self.model; del self.tokenizer; torch.cuda.empty_cache(); gc.collect()

# 6. 主程序
# ==========================================
def main():
    logger.info("🚀 启动评测 (HF + Local 混合模式)...")
    loader = HybridDataLoader()
    tasks = loader.load_all()

    if not tasks:
        logger.error("❌ 无有效数据。请检查网络或在 ./datasets 放入数据包。")
        return

    results = {}
    for m_name, conf in CONFIG['models'].items():
        if not conf['enabled']: continue
        logger.info(f"\n{'='*50}\n▶️  Model: {m_name}\n{'='*50}")
        try: evaluator = APIEvaluator(conf) if conf['type']=='api' else LocalEvaluator(conf)
        except Exception as e: logger.error(e); continue

        scores = {}
        for bench_name, dataset in tasks:
            total = 0
            for i in tqdm(range(0, len(dataset), conf['batch_size']), desc=f"{bench_name[:10]}", colour=conf['color']):
                batch = dataset[i:i+conf['batch_size']]
                preds = evaluator.generate([b['prompt'] for b in batch])
                total += sum([Scorer.score(p, b['gold']) for p,b in zip(preds, batch)]) * 100
            scores[bench_name] = total/len(dataset)
            logger.info(f"   🎯 {bench_name}: {scores[bench_name]:.2f}")

        results[m_name] = scores
        evaluator.unload()

    if results:
        df = pd.DataFrame(results).round(2)
        print("\n", df)
        df.to_excel(f"hybrid_results_{datetime.now().strftime('%H%M')}.xlsx")
        logger.info("✅ 评测完成。")

if __name__ == "__main__":
    main()
